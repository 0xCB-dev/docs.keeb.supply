{{- /* Custom section-menu override. Instructions section: - When inside a keyboard (e.g. /instructions/bad-wings-v2/…): show only that keyboard's section + its sub-pages. - When at the root /instructions/ page (list view): show all keyboards (they start collapsed via sidebar.collapsed). All other sections (basics, troubleshooting): Default Doks behaviour — sub-sections are collapsed by frontmatter param. */}} {{- $instrSection := .Site.GetPage "section" "instructions" }} {{- if eq .Section "instructions" }} {{- if or (.Eq $instrSection) (not .CurrentSection.Parent) (eq .CurrentSection $instrSection) }} {{- /* At the /instructions/ root — show all keyboards */}} {{ with $instrSection.Sections }} {{ partial "sidebar/render-section-menu.html" (dict "currentPage" $ "nodes" .) }} {{ end }} {{- else }} {{- /* Inside a keyboard: find the direct keyboard child of instructions */}} {{- $keyboard := .CurrentSection }} {{- if ne .CurrentSection.Parent $instrSection }} {{- $keyboard = .CurrentSection.Parent }} {{- end }} {{- /* Wrap in a slice so render-section-menu treats it as one top-level node */}} {{ with slice $keyboard }} {{ partial "sidebar/render-section-menu.html" (dict "currentPage" $ "nodes" .) }} {{ end }} {{- end }} {{- else }} {{- /* Basics, troubleshooting — default Doks section menu */}} {{- $sidebar := printf "sidebar_%s" .Section }} {{- with index site.Menus $sidebar }} {{ partial "sidebar/render-section-menu.html" (dict "currentPage" $ "nodes" .) }} {{- else }} {{ with (.Site.GetPage "section" .Section).Pages }} {{ partial "sidebar/render-section-menu.html" (dict "currentPage" $ "nodes" .) }} {{ end }} {{- end }} {{- end }}
